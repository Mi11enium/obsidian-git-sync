# Практическая Работа DOM
## Просмотр всех атрибутов элемента через document.getElementById
~~~html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <input type="text" id="test" value="one"> <!-- Вставляем поле с текстом "one" и id="test" -->
    <input type="button" onclick="func()"> <!-- Вставляем кнопку и по событию нажатия мыши (onclick) - вызываем функцию func() -->
</body>
</html>
<!-- Создаем функцию func(), в ней создаем переменную element и присваиваем ему значение gEBI == "test" -->
<!-- После присваивания значения мы можем вывести все значения атрибутов через alert(element.atribute) -->
<script>
    function func()
    {
        var element = document.getElementById("test");
        alert(element.value);
        alert(element.id);
        alert(element.type);
    }
</script>
~~~
